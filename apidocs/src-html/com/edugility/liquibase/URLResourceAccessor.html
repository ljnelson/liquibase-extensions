028package com.edugility.liquibase;
029
030import java.io.File; // for javadoc only
031import java.io.InputStream;
032import java.io.IOException;
033
034import java.net.MalformedURLException;
035import java.net.URL;
036
037import java.util.Collections;
038import java.util.Set;
039
040import liquibase.resource.AbstractResourceAccessor;
041import liquibase.resource.CompositeResourceAccessor; // for javadoc only
042import liquibase.resource.ResourceAccessor;
043
044/**
045 * An {@link AbstractResourceAccessor} that delegates most of its work
046 * to a delegate {@link ResourceAccessor}, while preserving the
047 * ability to process {@link String}s that represent {@link URL}s.
048 *
049 * <h2>Design Notes</h2>
050 *
051 * <p>The Liquibase 3.3.0 {@link ResourceAccessor} interface is
052 * inconsistently documented, and mixes several quite unrelated
053 * concerns together.  The {@link CompositeResourceAccessor}
054 * implicitly relies on non-{@code null} values being returned from
055 * some of its aggregated {@link ResourceAccessor}s' methods.  The
056 * combination of these two unfortunate facts means that returning
057 * {@code null} from methods that do not make sense for a {@link
058 * ResourceAccessor} like this one that processes URLs is effectively
059 * impossible.  Consequently, a delegating approach has to be taken
060 * instead.</p>
061 *
062 * @author <a href="http://about.me/lairdnelson"
063 * target="_parent">Laird Nelson</a>
064 *
065 * @see AbstractResourceAccessor
066 *
067 * @see CompositeResourceAccessor
068 *
069 * @see ResourceAccessor
070 */
071public class URLResourceAccessor extends AbstractResourceAccessor {
072
073  /**
074   * The non-{@code null} {@link ResourceAccessor} to which the {@link
075   * #list(String, String, boolean, boolean, boolean)} and {@link
076   * #toClassLoader()} methods are delegated.
077   *
078   * <p>This field is never {@code null}.</p>
079   *
080   * @see #URLResourceAccessor(ResourceAccessor)
081   */
082  private final ResourceAccessor delegate;
083
084  /**
085   * Creates a new {@link URLResourceAccessor}.
086   *
087   * @param delegate the {@link ResourceAccessor} to which the {@link
088   * #list(String, String, boolean, boolean, boolean)} and {@link
089   * #toClassLoader()} methods will be delegated; must not be {@code
090   * null}
091   *
092   * @exception IllegalArgumentException if {@code delegate} is {@code
093   * null}
094   */
095  public URLResourceAccessor(final ResourceAccessor delegate) {
096    super();
097    if (delegate == null) {
098      throw new IllegalArgumentException("delegate", new NullPointerException("delegate"));
099    }
100    this.delegate = delegate;
101    this.init();
102  }
103
104  /**
105   * Returns a {@link Set} of <strong>{@linkplain InputStream#close()
106   * open}</strong> {@link InputStream}s that the supplied {@code
107   * path} logically designates, or {@code null} if there is no
108   * resource that can be identified with the supplied {@code path},
109   * per the contract of the {@link
110   * ResourceAccessor#getResourcesAsStream(String)} method.
111   *
112   * <p>This method may return {@code null} if there is no resource
113   * that can be identified by the supplied {@code path}.</p>
114   *
115   * <p>This implementation attempts to {@linkplain URL#URL(String)
116   * create a <code>URL</code> from the supplied <code>path</code>},
117   * and, if that operation is successful, to return a {@linkplain
118   * Collections#singleton(Object) <code>Set</code> containing only}
119   * that {@code URL}.  If a {@link MalformedURLException} is
120   * encountered during this operation, then {@code null} is returned
121   * instead.</p>
122   *
123   * @param path a {@link String} that will be passed unaltered to the
124   * {@linkplain URL#URL(String) constructor of the <code>URL</code>
125   * class}; must not be {@code null}
126   *
127   * @return a {@link Set} of <strong>{@linkplain InputStream#close()
128   * open}</strong> {@link InputStream}s, or {@code null}
129   *
130   * @exception NullPointerException if {@code path} is {@code null}
131   *
132   * @exception IOException if an input/output error occurs
133   *
134   * @see ResourceAccessor#getResourcesAsStream(String)
135   */
136  @Override
137  public Set<InputStream> getResourcesAsStream(final String path) throws IOException {
138    final Set<InputStream> returnValue;
139    if (path == null) {
140      returnValue = null;
141    } else {
142      Set<InputStream> temp = null;
143      try {
144        temp = Collections.singleton(new URL(path).openStream());
145      } catch (final MalformedURLException returnNullInsteadPerContract) {
146        temp = null;
147      } finally {
148        returnValue = temp;
149      }
150    }
151    return returnValue;
152  }
153
154  /**
155   * Returns {@code null} when invoked.
156   * 
157   * <p>The {@link ResourceAccessor} interface does not document all
158   * of the parameters to this method.  They are passed to the
159   * delegate unaltered.</p>
160   *
161   * <p>The {@link ResourceAccessor} interface does not document the
162   * purpose of this method that subclasses are required to implement.
163   * Consequently the behavior of this implementation may be
164   * incorrect.  Use it with care.</p>
165   *
166   * @param relativeTo a {@link String} passed to the {@link
167   * ResourceAccessor#list(String, String, boolean, boolean, boolean)}
168   * method on the {@link ResourceAccessor} supplied at {@linkplain
169   * #URLResourceAccessor(ResourceAccessor) construction time}
170   * unaltered
171   *
172   * @param path a {@link String} passed to the {@link
173   * ResourceAccessor#list(String, String, boolean, boolean, boolean)}
174   * method on the {@link ResourceAccessor} supplied at {@linkplain
175   * #URLResourceAccessor(ResourceAccessor) construction time}
176   *
177   * @param includeFiles whether non-directory {@link File}s should be
178   * returned; passed to the {@link ResourceAccessor#list(String,
179   * String, boolean, boolean, boolean)} method on the {@link
180   * ResourceAccessor} supplied at {@linkplain
181   * #URLResourceAccessor(ResourceAccessor) construction time}
182   *
183   * @param includeDirectories whether {@linkplain File#isDirectory()
184   * directories} should be returned; passed to the {@link
185   * ResourceAccessor#list(String, String, boolean, boolean, boolean)}
186   * method on the {@link ResourceAccessor} supplied at {@linkplain
187   * #URLResourceAccessor(ResourceAccessor) construction time}
188   *
189   * @param recursive if {@code true}, a {@code boolean} passed to the {@link
190   * ResourceAccessor#list(String, String, boolean, boolean, boolean)}
191   * method on the {@link ResourceAccessor} supplied at {@linkplain
192   * #URLResourceAccessor(ResourceAccessor) construction time}
193   *
194   * @return {@code null} when invoked
195   *
196   * @exception IOException if an error occurs
197   *
198   * @see ResourceAccessor#list(String, String, boolean, boolean, boolean)
199   */
200  @Override
201  public Set<String> list(final String relativeTo, final String path, final boolean includeFiles, final boolean includeDirectories, final boolean recursive) throws IOException {
202    return null;
203  }
204
205  /**
206   * Returns the result of invoking the {@link
207   * ResourceAccessor#toClassLoader()} method on the {@link
208   * ResourceAccessor} supplied at {@linkplain
209   * #URLResourceAccessor(ResourceAccessor) construction time}.
210   *
211   * @return a {@link ClassLoader}, or {@code null}
212   *
213   * @see ResourceAccessor#toClassLoader()
214   */
215  @Override
216  public ClassLoader toClassLoader() {
217    if (this.delegate == null) {
218      return null;
219    } else {
220      return this.delegate.toClassLoader();
221    }
222  }
223
