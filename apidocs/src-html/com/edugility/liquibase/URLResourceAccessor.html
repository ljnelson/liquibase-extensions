028package com.edugility.liquibase;
029
030import java.io.File; // for javadoc only
031import java.io.InputStream;
032import java.io.IOException;
033
034import java.net.MalformedURLException;
035import java.net.URL;
036
037import java.util.Collections;
038import java.util.Set;
039
040import liquibase.resource.AbstractResourceAccessor;
041import liquibase.resource.CompositeResourceAccessor; // for javadoc only
042import liquibase.resource.ResourceAccessor;
043
044/**
045 * An {@link AbstractResourceAccessor} that delegates most of its work
046 * to a delegate {@link ResourceAccessor}, while preserving the
047 * ability to process {@link String}s that represent {@link URL}s.
048 *
049 * <h2>Design Notes</h2>
050 *
051 * <p>The Liquibase 3.3.0 {@link ResourceAccessor} interface is
052 * inconsistently documented, and mixes several quite unrelated
053 * concerns together.  The {@link CompositeResourceAccessor}
054 * implicitly relies on non-{@code null} values being returned from
055 * some of its aggregated {@link ResourceAccessor}s' methods.  The
056 * combination of these two unfortunate facts means that returning
057 * {@code null} from methods that do not make sense for a {@link
058 * ResourceAccessor} like this one that processes URLs is effectively
059 * impossible.  Consequently, a delegating approach has to be taken
060 * instead.</p>
061 *
062 * @author <a href="http://about.me/lairdnelson"
063 * target="_parent">Laird Nelson</a>
064 *
065 * @see AbstractResourceAccessor
066 *
067 * @see CompositeResourceAccessor
068 *
069 * @see ResourceAccessor
070 */
071public class URLResourceAccessor extends AbstractResourceAccessor {
072
073  /**
074   * The non-{@code null} {@link ResourceAccessor} to which the {@link
075   * #toClassLoader()} method is delegated.
076   *
077   * <p>This field is never {@code null}.</p>
078   *
079   * @see #URLResourceAccessor(ResourceAccessor)
080   */
081  private final ResourceAccessor delegate;
082
083  /**
084   * Creates a new {@link URLResourceAccessor}.
085   *
086   * @param delegate the {@link ResourceAccessor} to which the {@link
087   * #toClassLoader()} method will be delegated; must not be {@code
088   * null}
089   *
090   * @exception IllegalArgumentException if {@code delegate} is {@code
091   * null}
092   */
093  public URLResourceAccessor(final ResourceAccessor delegate) {
094    super();
095    if (delegate == null) {
096      throw new IllegalArgumentException("delegate", new NullPointerException("delegate"));
097    }
098    this.delegate = delegate;
099    this.init();
100  }
101
102  /**
103   * Returns a {@link Set} of <strong>{@linkplain InputStream#close()
104   * open}</strong> {@link InputStream}s that the supplied {@code
105   * path} logically designates, or {@code null} if there is no
106   * resource that can be identified with the supplied {@code path},
107   * per the contract of the {@link
108   * ResourceAccessor#getResourcesAsStream(String)} method.
109   *
110   * <p>This method may return {@code null} if there is no resource
111   * that can be identified by the supplied {@code path}.</p>
112   *
113   * <p>This implementation attempts to {@linkplain URL#URL(String)
114   * create a <code>URL</code> from the supplied <code>path</code>},
115   * and, if that operation is successful, to return a {@linkplain
116   * Collections#singleton(Object) <code>Set</code> containing only}
117   * that {@code URL}.  If a {@link MalformedURLException} is
118   * encountered during this operation, then {@code null} is returned
119   * instead.</p>
120   *
121   * @param path a {@link String} that will be passed unaltered to the
122   * {@linkplain URL#URL(String) constructor of the <code>URL</code>
123   * class}; must not be {@code null}
124   *
125   * @return a {@link Set} of <strong>{@linkplain InputStream#close()
126   * open}</strong> {@link InputStream}s, or {@code null}
127   *
128   * @exception NullPointerException if {@code path} is {@code null}
129   *
130   * @exception IOException if an input/output error occurs
131   *
132   * @see ResourceAccessor#getResourcesAsStream(String)
133   */
134  @Override
135  public Set<InputStream> getResourcesAsStream(final String path) throws IOException {
136    final Set<InputStream> returnValue;
137    if (path == null) {
138      returnValue = null;
139    } else {
140      Set<InputStream> temp = null;
141      try {
142        temp = Collections.singleton(new URL(path).openStream());
143      } catch (final MalformedURLException returnNullInsteadPerContract) {
144        temp = null;
145      } finally {
146        returnValue = temp;
147      }
148    }
149    return returnValue;
150  }
151
152  /**
153   * Returns {@code null} when invoked.
154   *
155   * <p>The {@link ResourceAccessor} interface does not document the
156   * purpose of this method that subclasses are required to implement
157   * or all of its parameters.  Consequently the behavior of this
158   * implementation may be incorrect.  Use it with care.</p>
159   *
160   * @param relativeTo ignored
161   *
162   * @param path ignored
163   *
164   * @param includeFiles ignored
165   *
166   * @param includeDirectories ignored
167   *
168   * @param recursive ignored
169   *
170   * @return {@code null} when invoked
171   *
172   * @exception IOException if an error occurs
173   *
174   * @see ResourceAccessor#list(String, String, boolean, boolean, boolean)
175   */
176  @Override
177  public Set<String> list(final String relativeTo, final String path, final boolean includeFiles, final boolean includeDirectories, final boolean recursive) throws IOException {
178    return null;
179  }
180
181  /**
182   * Returns the result of invoking the {@link
183   * ResourceAccessor#toClassLoader()} method on the {@link
184   * ResourceAccessor} supplied at {@linkplain
185   * #URLResourceAccessor(ResourceAccessor) construction time}.
186   *
187   * @return a {@link ClassLoader}, or {@code null}
188   *
189   * @see ResourceAccessor#toClassLoader()
190   */
191  @Override
192  public ClassLoader toClassLoader() {
193    if (this.delegate == null) {
194      return null;
195    } else {
196      return this.delegate.toClassLoader();
197    }
198  }
199  
200}
