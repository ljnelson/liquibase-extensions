028package com.edugility.liquibase;
029
030import java.io.InputStream;
031import java.io.IOException;
032
033import java.net.MalformedURLException;
034import java.net.URL;
035
036import java.util.Collections;
037import java.util.Set;
038
039import liquibase.resource.AbstractResourceAccessor;
040import liquibase.resource.CompositeResourceAccessor; // for javadoc only
041import liquibase.resource.ResourceAccessor;
042
043/**
044 * An {@link AbstractResourceAccessor} that delegates most of its work
045 * to a delegate {@link ResourceAccessor}, while preserving the
046 * ability to process {@link String}s that represent {@link URL}s.
047 *
048 * <h2>Design Notes</h2>
049 *
050 * <p>The Liquibase 3.2.2 {@link ResourceAccessor} interface is
051 * inconsistently documented, and mixes several quite unrelated
052 * concerns together.  The {@link CompositeResourceAccessor}
053 * implicitly relies on non-{@code null} values being returned from
054 * some of its aggregated {@link ResourceAccessor}s' methods.  The
055 * combination of these two unfortunate facts means that returning
056 * {@code null} from methods that do not make sense for a {@link
057 * ResourceAccessor} like this one that processes URLs is effectively
058 * impossible.  Consequently, a delegating approach has to be taken
059 * instead.</p>
060 *
061 * @author <a href="http://about.me/lairdnelson"
062 * target="_parent">Laird Nelson</a>
063 *
064 * @see AbstractResourceAccessor
065 *
066 * @see CompositeResourceAccessor
067 *
068 * @see ResourceAccessor
069 */
070public class URLResourceAccessor extends AbstractResourceAccessor {
071
072  /**
073   * The non-{@code null} {@link ResourceAccessor} to which the {@link
074   * #list(String, String, boolean, boolean, boolean)} and {@link
075   * #toClassLoader()} methods are delegated.
076   *
077   * <p>This field is never {@code null}.</p>
078   *
079   * @see #URLResourceAccessor(ResourceAccessor)
080   */
081  private final ResourceAccessor delegate;
082
083  /**
084   * Creates a new {@link URLResourceAccessor}.
085   *
086   * @param delegate the {@link ResourceAccessor} to which the {@link
087   * #list(String, String, boolean, boolean, boolean)} and {@link
088   * #toClassLoader()} methods will be delegated; must not be {@code
089   * null}
090   *
091   * @exception IllegalArgumentException if {@code delegate} is {@code
092   * null}
093   */
094  public URLResourceAccessor(final ResourceAccessor delegate) {
095    super();
096    if (delegate == null) {
097      throw new IllegalArgumentException("delegate", new NullPointerException("delegate"));
098    }
099    this.delegate = delegate;
100  }
101
102  /**
103   * Returns a {@link Set} of [@linkplain InputStream#close() open}
104   * {@link InputStream}s that the supplied {@code path} logically
105   * designates.
106   *
107   * <p>This method never returns {@code null}.</p>
108   *
109   * <p>This implementation attempts to {@linkplain URL#URL(String)
110   * create a <code>URL</code> from the supplied <code>path</code>},
111   * and, if that operation is successful, to return a {@linkplain
112   * Collections#singleton(Object) <code>Set</code> containing
113   * only} that {@code URL}.  If a {@link MalformedURLException} is
114   * encountered during this operation, then the return value of the
115   * {@link #getResourcesAsStream(String)} method invoked on the
116   * {@linkplain #URLResourceAccessor(ResourceAccessor) delegate} is
117   * returned instead.</p>
118   *
119   * <p>If a {@code null} return value is about to be returned for any
120   * reason, then the return value of the {@link
121   * Collections#emptySet()} method is returned instead.</p>
122   *
123   * @param path a {@link String} that will be passed unaltered to the
124   * {@linkplain URL#URL(String) constructor of the <code>URL</code>
125   * class}; must not be {@code null}
126   *
127   * @return a non-{@code null} {@link Set} of open {@link InputStream}s
128   *
129   * @exception NullPointerException if {@code path} is {@code null}
130   *
131   * @exception IOException if an input/output error occurs
132   */
133  @Override
134  public Set<InputStream> getResourcesAsStream(final String path) throws IOException {
135    Set<InputStream> returnValue = null;
136    if (path != null) {
137      try {
138        returnValue = Collections.singleton(new URL(path).openStream());
139      } catch (final MalformedURLException expectedInSomeCases) {
140        returnValue = this.delegate.getResourcesAsStream(path);
141      }
142    }
143    if (returnValue == null || returnValue.isEmpty()) {
144      returnValue = Collections.emptySet();
145    }
146    return returnValue;
147  }
148
149  /**
150   * Returns the result of invoking the {@link
151   * ResourceAccessor#list(String, String, boolean, boolean, boolean)}
152   * method on the {@link ResourceAccessor} supplied at {@linkplain
153   * #URLResourceAccessor(ResourceAccessor) construction time}.
154   *
155   * <p>The {@link ResourceAccessor} interface does not document any
156   * of the parameters to this method.  They are passed to the
157   * delegate unaltered.</p>
158   *
159   * @param relativeTo a {@link String} passed to the {@link
160   * ResourceAccessor#list(String, String, boolean, boolean, boolean)}
161   * method on the {@link ResourceAccessor} supplied at {@linkplain
162   * #URLResourceAccessor(ResourceAccessor) construction time}
163   * unaltered
164   *
165   * @param path a {@link String} passed to the {@link
166   * ResourceAccessor#list(String, String, boolean, boolean, boolean)}
167   * method on the {@link ResourceAccessor} supplied at {@linkplain
168   * #URLResourceAccessor(ResourceAccessor) construction time}
169   *
170   * @param includeFiles a {@code boolean} passed to the {@link
171   * ResourceAccessor#list(String, String, boolean, boolean, boolean)}
172   * method on the {@link ResourceAccessor} supplied at {@linkplain
173   * #URLResourceAccessor(ResourceAccessor) construction time}
174   *
175   * @param includeDirectories a {@code boolean} passed to the {@link
176   * ResourceAccessor#list(String, String, boolean, boolean, boolean)}
177   * method on the {@link ResourceAccessor} supplied at {@linkplain
178   * #URLResourceAccessor(ResourceAccessor) construction time}
179   *
180   * @param recursive a {@code boolean} passed to the {@link
181   * ResourceAccessor#list(String, String, boolean, boolean, boolean)}
182   * method on the {@link ResourceAccessor} supplied at {@linkplain
183   * #URLResourceAccessor(ResourceAccessor) construction time}
184   *
185   * @return a {@link Set} of {@link String}s, or {@code null}
186   *
187   * @exception IOException if an error occurs
188   *
189   * @see ResourceAccessor#list(String, String, boolean, boolean, boolean)
190   */
191  @Override
192  public Set<String> list(final String relativeTo, final String path, final boolean includeFiles, final boolean includeDirectories, final boolean recursive) throws IOException {
193    return this.delegate.list(relativeTo, path, includeFiles, includeDirectories, recursive);
194  }
195
196  /**
197   * Returns the result of invoking the {@link
198   * ResourceAccessor#toClassLoader()} method on the {@link
199   * ResourceAccessor} supplied at {@linkplain
200   * #URLResourceAccessor(ResourceAccessor) construction time}.
201   *
202   * @return a {@link ClassLoader}, or {@code null}
203   *
204   * @see ResourceAccessor#toClassLoader()
205   */
206  @Override
207  public ClassLoader toClassLoader() {
208    return this.delegate.toClassLoader();
209  }
210  
211}
